161  private <T> Response executeGetRequest(Map<String, String> routeParamsMap, Map<String, String> queryStringMap, Map<String, String> headersMap, Class<T> clazz, GetRequest request) throws HttpClientException {
162  if (user != null && password != null) {
163  request.basicAuth(user, password);
164  }
165 
166  if(routeParamsMap != null) {
167  for (Map.Entry<String, String> entry : routeParamsMap.entrySet()) {
168  request.routeParam(entry.getKey(), entry.getValue());
169  }
170  }
171  if (queryStringMap != null) {
172  for (Map.Entry<String, String> entry : queryStringMap.entrySet()) {
173  request.queryString(entry.getKey(), entry.getValue());
174  }
175  }
176  if (headersMap != null) {
177  for (Map.Entry<String, String> entry : headersMap.entrySet()) {
178  request.header(entry.getKey(), entry.getValue());
179  }
180  }
181  try {
182  HttpResponse<T> response = request.asObject(clazz);
183  Response<T> r = new Response();
184  r.setStatusCode(response.getStatus());
185  r.setStatusText(response.getStatusText());
186  r.setRawBody(getStringFromInputStream(response.getRawBody()));
187  r.setParsedObject(response.getBody());
188  return r;
189  } catch (UnirestException e) {
190  throw new HttpClientException(e);
191  }
192  }

196  if (user != null && password != null) {
