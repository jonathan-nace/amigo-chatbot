43  public String execute(String dockerImage, List<String> envList, List<String> commandList, String entryPoint) throws CommandExecutionException {
44  try( DockerClient dockerClient = new DefaultDockerClient("unix:///var/run/docker.sock")) {
45  String response = null;

59  // Wait for the container to exit.
60  // If we don't wait, docker.logs() might return an epmty string because the container

63 
64  final String log;
65  try (LogStream logs = dockerClient.logs(containerId, stdout(), stderr())) {
66  log = logs.readFully();
67  }
68  logger.info(log);
69  response = log;
70  dockerClient.removeContainer(containerId);
