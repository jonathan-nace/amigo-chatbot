52  private Optional<PrincipalUser> getUserByCredentials(String username, String password) {
53  UserDAO userDAO = (UserDAO) dbClient.getDAO(UserDAO.class);
54  PrincipalUser principalUser = null;
55 
56  try {
57  List<User> users = userDAO.fetchById(new ArrayList<String>() {{
58  add(username);
59  }});
60  if (users != null && !users.isEmpty()) {
61  User user = users.get(0);
62  if (user.getPassword().equals(Utilities.generateMD5Hash(password))) {
63  principalUser = new PrincipalUser(username);
64  principalUser.setPassword(password);
65  return Optional.of(principalUser);
66  }
67  }
68  } catch (DBException | NoSuchAlgorithmException e) {
69  log.log(Level.SEVERE, MessageFormat.format("Authentication failed for user {0}", username), e);
70  }
